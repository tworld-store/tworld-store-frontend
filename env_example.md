
이 파일은 env.example의 실제 코드임.
.example 파일 업로드 불가에 따라 마크다운으로 만들어 놓은 것.


# Firebase 설정
VITE_FIREBASE_API_KEY=your-api-key
VITE_FIREBASE_AUTH_DOMAIN=your-project.firebaseapp.com
VITE_FIREBASE_PROJECT_ID=your-project-id
VITE_FIREBASE_STORAGE_BUCKET=your-project.appspot.com
VITE_FIREBASE_MESSAGING_SENDER_ID=123456789
VITE_FIREBASE_APP_ID=1:123456789:web:abcdef

# Cloudinary 설정
VITE_CLOUDINARY_CLOUD_NAME=your-cloud-name
VITE_CLOUDINARY_API_KEY=your-api-key
VITE_CLOUDINARY_API_SECRET=your-api-secret

# Google Sheets 설정
GOOGLE_SHEETS_ID=your-google-sheets-id
GOOGLE_SERVICE_ACCOUNT_EMAIL=your-service-account@project.iam.gserviceaccount.com

# Telegram Bot 설정
TELEGRAM_BOT_TOKEN=your-bot-token
TELEGRAM_CHAT_ID=your-chat-id

# products.json URL (GitHub Raw)
VITE_PRODUCTS_JSON_URL=https://raw.githubusercontent.com/tworld-store-frontend/tworld-store-frontend/main/data/products.json

# API 설정
API_BASE_URL=https://your-domain.com/api
CDN_URL=https://cdn.your-domain.com

# 개발 모드
NODE_ENV=development
